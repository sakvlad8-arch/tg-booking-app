    <!-- ЭКРАН 1: Выбор услуги -->

        </div>

        <div onclick="selectService('Функциональный тренинг', '25$')" class="border rounded-xl p-4 cursor-pointer hover:border-blue-500 transition-all bg-[var(--tg-theme-secondary-bg-color,#f4f4f5)]">

        <h2 class="text-sm font-semibold uppercase tracking-wider opacity-60">Выберите дату и время:</h2>
        
        <!-- Простая имитация дней -->
        <div class="flex gap-2 overflow-x-auto pb-2">
            <button onclick="selectDate('Сегодня')" class="border rounded-lg px-4 py-2 text-sm bg-blue-500 text-white font-medium">Сегодня</button>

        <!-- Свободные слоты по 45 минут -->
        <div class="grid grid-cols-3 gap-2 mt-4">

    <!-- ЭКРАН 3: Подтверждение (скрыт по умолчанию) -->
    <div id="step-confirm" class="hidden space-y-4">
        <button onclick="goBack('time')" class="text-sm text-[var(--tg-theme-link-color,#2481cc)] mb-2">← Назад к времени</button>
        <h2 class="text-sm font-semibold uppercase tracking-wider opacity-60">Проверьте данные:</h2>
        
        <div class="bg-[var(--tg-theme-secondary-bg-color,#f4f4f5)] p-4 rounded-xl space-y-1">
            <div><strong>Услуга:</strong> <span id="summary-service">-</span></div>
            <div><strong>Дата:</strong> <span id="summary-date">Сегодня</span></div>
            <div><strong>Время:</strong> <span id="summary-time">-</span></div>
        </div>

        <button onclick="submitBooking()" class="btn-primary w-full py-3 rounded-xl font-bold mt-4 shadow-md">
            Подтвердить запись
        </button>

    <script>
        // Инициализируем Telegram Web App
        const tg = window.Telegram.WebApp;
        tg.ready(); // Сообщаем ТГ, что приложение загрузилось
        tg.expand(); // Разворачиваем на весь экран

        // Объект, куда будем собирать данные записи
        let bookingData = {
            service: '',
            price: '',
            date: 'Сегодня',
            time: ''
        };

        function selectService(name, price) {
            bookingData.service = name;
            bookingData.price = price;
            
            document.getElementById('step-services').classList.add('hidden');
            document.getElementById('step-time').classList.remove('hidden');
        }

        function selectDate(date) {
            bookingData.date = date;
            // Здесь в будущем добавим визуальное переключение активного дня
        }

        function selectTime(time) {
            bookingData.time = time;
            
            document.getElementById('summary-service').innerText = `${bookingData.service} (${bookingData.price})`;
            document.getElementById('summary-date').innerText = bookingData.date;
            document.getElementById('summary-time').innerText = bookingData.time + ' (45 мин)';

            document.getElementById('step-time').classList.add('hidden');
            document.getElementById('step-confirm').classList.remove('hidden');
        }

        function goBack(toStep) {
            document.getElementById('step-services').classList.add('hidden');
            document.getElementById('step-time').classList.add('hidden');
            document.getElementById('step-confirm').classList.add('hidden');

            document.getElementById(`step-${toStep}`).classList.remove('hidden');
        }

        function submitBooking() {
            // Отправляем данные обратно боту в чат в формате JSON
            tg.sendData(JSON.stringify(bookingData));
            // Закрываем мини-приложение
            tg.close();
        }
    </script>
